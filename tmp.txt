datasource db {
  provider = "postgresql"
}

enum UserType {
  Provider,
  Client
}

model User {
  id            Int @id @default(autoincrement())
  userType      UserType
}

model Availability {
  start         DateTime
  end           DateTime
  provider      user @relation(fields: [providerID], references[id])
  providerID    Int
}

model Reservation {
  time          DateTime
  expiration    DateTime
  confirmed     Boolean
  client        User @relation(fields: [clientID], references[id])
  provider      User @relation(fields: [providerID], references[id])
  clientID      Int
  providerID    Int
}
